function(icuMain)
  set(library_root "${CMAKE_CURRENT_SOURCE_DIR}/src/icu4c/source")

  add_library(thirdparty_icu INTERFACE)

  target_link_libraries(thirdparty_icu INTERFACE
    global_cxx_settings
  )

  target_include_directories(thirdparty_icu SYSTEM INTERFACE
    "${library_root}/i18n"
    "${library_root}/common"
  )
endfunction()

icuMain()
